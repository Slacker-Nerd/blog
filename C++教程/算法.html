{

    int min = a[0], max = a[0];
    for (int i = 0; i < len; i++)

// 冒泡排序
void BubbleSort(int *a, int len)
{
    for (int i = 0; i < len - 1; i++)
    {
        for (int i = 0; i < len - 1; i++)
        {
            if (a[i] > a[i + 1])
            {
                int t = a[i];
                a[i] = a[i + 1];
                a[i + 1] = t;
            }
        }
    }
}

// 选择排序
void SelectSort(int *a, int len)
{
    int b;
    for (int j = 0; j < len; j++)
    {
        b = j;
        for (int i = j; i < len; i++)
        {
            if (a[i] < a[b])
            {
                b = i;
            }
        }
        if (b != j)
        {
            int t = a[b];
            a[b] = a[j];
            a[j] = t;
        }
    }
}

// 插入排序
void InsertSort(int *a, int len)
{
    for (int i = 1; i < len; i++)
    {
        if (a[i] < a[i - 1])
        {
            int tmp = a[i];
            int j;
            for (j = i - 1; j >= 0 && a[j] > tmp; j--)
            {
                a[j + 1] = a[j];
            }
            a[j + 1] = tmp;
        }
    }
}

# 等差数列

定义：相邻两个数的差值恒定

例子：100,102,104,106

设：首项：$$a_1$$、公差：$d$、项数：$n$ $$ a_n=a_1+(n-1)d $$

$$ S_n=\frac{(a_1+a_n)n}{2}=\frac{(a_1+a_1+(n-1)d)n}{2}=na_1+\frac{n(n-1)d}{2} $$

# 等比数列

定义：相邻两个数的差值恒定

例子：6,12,24,48

设：首项：$$a_1$$、公差：$q$、项数：$n$ $$ a_n=a_1q^{n-1} $$

$$ S_n=\frac{1-q^n}{1-q}*a_1 $$
